Zig Standard Library

std

Description: The root of Zig’s standard library, providing core language facilities and modules.

Syntax

const std = @import("std");

Example

const std = @import("std");

pub fn main() void {
    std.debug.print("Hello from std!\n", .{});
}

annotation

Description: Provides metadata or annotation utilities for language structures.

Syntax

const std = @import("std");
const annotation = std.annotation;

Example

// Hypothetical usage:
// annotation.someFunction(value);

const std = @import("std");

Example

fn exampleFn(comptime T: anytype) void {
    // ...
}

array_hash_map

Description: A hash map stored with arrays, providing key-value storage.

Syntax

const std = @import("std");
const ArrayHashMap = std.array_hash_map.ArrayHashMap;

Example

const std = @import("std");

pub fn main() !void {
    var map = ArrayHashMap(u32, bool).init(std.heap.page_allocator);
    try map.put(42, true);
    const val = map.get(42) orelse false;
    std.debug.print("Value: {}\n", .{val});
}

pub fn main() !void {
    var list = ArrayList(u8).init(std.heap.page_allocator);
    try list.append(97); // 'a'
    std.debug.print("First item: {c}\n", .{list.items[0]});
}
